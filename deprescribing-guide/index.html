function escapeRegex(str) {

  return str.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');

}


function highlight(text, query) {

  if (!query) return text;

  const escaped = escapeRegex(query);

  const regex = new RegExp(`(${escaped})`, 'gi');

  return text.replace(regex, '<mark class="highlight">$1</mark>');

}


function searchMedicines(query) {

if (!query || query.trim().length < 2) return [];

const terms = query.trim().toLowerCase().replace(/\s+/g, ' ').split(' ').filter(t => t.length >= 2);

if (terms.length === 0) return [];

return deprescribingData.filter(d => {

const haystack = [d.medicine||'',d.category||'',d.keywords||'',d.when||'',d.how||'',d.monitoring||'',d.evidence||''].join(' ').toLowerCase();

return terms.every(term => haystack.includes(term));

});

}
